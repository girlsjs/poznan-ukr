–î–µ—è–∫—ñ –ø–æ–¥—ñ—ó –≤—ñ–¥–±—É–≤–∞—é—Ç—å—Å—è –ª–∏—à–µ —Ç–æ–¥—ñ, –∫–æ–ª–∏ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è –ø–µ–≤–Ω–∞ —É–º–æ–≤–∞. –ù–∞–ø—Ä–∏–∫–ª–∞–¥, –≤–æ–¥–∞ –∑–∞–∫–∏–ø–∏—Ç—å, –∫–æ–ª–∏ –¥–æ—Å—è–≥–Ω–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∏ 100 –≥—Ä–∞–¥—É—Å—ñ–≤, –∞ –¥–≤–µ—Ä—ñ –º–æ–∂–Ω–∞ –≤—ñ–¥—á–∏–Ω–∏—Ç–∏, —Ç—ñ–ª—å–∫–∏ —è–∫—â–æ —É –≤–∞—Å —î –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–∏–π –∫–ª—é—á. –¢–µ –∂ —Å–∞–º–µ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è —ñ –≤ JavaScript: —É–º–æ–≤–Ω–∏–π –æ–ø–µ—Ä–∞—Ç–æ—Ä –≤–∏–∫–æ–Ω—É—î –∫–æ–¥ –Ω–∞ –≤–∞—à –≤–∏–±—ñ—Ä, –∑–∞–ª–µ–∂–Ω–æ –≤—ñ–¥ —Ç–æ–≥–æ, —á–∏ —î –∑–Ω–∞—á–µ–Ω–Ω—è –≤–∏—Ä–∞–∑—É –ª–æ–≥—ñ—á–Ω–æ—é —ñ—Å—Ç–∏–Ω–æ—é \(true\) –∞–±–æ –ª–æ–≥—ñ—á–Ω–æ—é —Ö–∏–±–æ—é \(false\).

## if... else...

–Ø–∫ –≤–∏–≥–ª—è–¥–∞—î —É–º–æ–≤–Ω–∞ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ü—ñ—è? –ö–ª—é—á–æ–≤–∏–º –µ–ª–µ–º–µ–Ω—Ç–æ–º —î –æ–ø–µ—Ä–∞—Ç–æ—Ä **if**:

```js

    ...–∫–æ–º–∞–Ω–¥–∞, —è–∫–∞ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è –ø—Ä–∏ –≤–∏–∫–æ–Ω–∞–Ω–Ω—ñ –ø–µ–≤–Ω–æ—ó —É–º–æ–≤–∏
}
```

–ù–∞–ø—Ä–∏–∫–ª–∞–¥:

```js

if (x < 100) {
    console.log('Liczba jest mniejsza od 100');
}
```

```js

     console.log('Liczba jest mniejsza od 10');
 }
```

–ù–∞—Å—Ç—É–ø–Ω–∏–º –µ–ª–µ–º–µ–Ω—Ç–æ–º —î —É–º–æ–≤–Ω–∏–π –æ–ø–µ—Ä–∞—Ç–æ—Ä **else**, —Ç–æ–±—Ç–æ –∫–æ–º–∞–Ω–¥–∞, —è–∫–∞ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è, —è–∫—â–æ —É–º–æ–≤–∞ –Ω–µ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è.

–ù–∞–ø—Ä–∏–∫–ª–∞–¥:

```js

if (x < 100) {
     console.log('–ß–∏—Å–ª–æ –º–µ–Ω—à–µ 100');

     console.log('–ß–∏—Å–ª–æ –±—ñ–ª—å—à–µ 100');

     console.log('–ß–∏—Å–ª–æ –º–µ–Ω—å—à–µ 10');

     console.log('–ß–∏—Å–ª–æ –±—ñ–ª—å—à–µ 10');
```

–ú–∏ —Ç–∞–∫–æ–∂ –º–æ–∂–µ–º–æ –ø–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏ –∫—ñ–ª—å–∫–∞ —É–º–æ–≤ –æ–¥–Ω–∞ –∑–∞ –æ–¥–Ω–æ—é. –î–ª—è —Ü—å–æ–≥–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è **else if**:

```js
     console.log('–ß–∏—Å–ª–æ –¥–æ—Ä—ñ–≤–Ω—é—î 10');
```

–°—Ç–≤–æ—Ä–∏ —É —Å–≤–æ—î–º—É JS-—Ñ–∞–π–ª—ñ –¥–≤—ñ –∑–º—ñ–Ω–Ω—ñ –∑ —ñ–º–µ–Ω–∞–º–∏ a —Ç–∞ b. –ü—Ä–∏—Å–≤–æ–π —ó–º –¥–≤–∞ —Ä—ñ–∑–Ω–∏—Ö —á–∏—Å–ª–∞. –ü–æ—Ç—ñ–º –Ω–∞–ø–∏—à–∏ –Ω–∞—Å—Ç—É–ø–Ω—É —É–º–æ–≤—É: —è–∫—â–æ a –±—ñ–ª—å—à–µ b, —Ç–æ –≤ –∫–æ–Ω—Å–æ–ª—ñ –ø–æ–≤–∏–Ω–Ω–æ –±—É—Ç–∏ –≤–∏–≤–µ–¥–µ–Ω–æ "a –±—ñ–ª—å—à–µ b". –Ø–∫—â–æ b –±—ñ–ª—å—à–µ, –≤ –∫–æ–Ω—Å–æ–ª—ñ –ø–æ–≤–∏–Ω–Ω–æ –±—É—Ç–∏ –≤–∏–≤–µ–¥–µ–Ω–æ "b –±—ñ–ª—å—à–µ –∑–∞ a".

### switch

**–£–º–æ–≤–∏ —Ç–∞–∫–æ–∂ –º–æ–∂–Ω–∞ –ø–µ—Ä–µ–≤—ñ—Ä—è—Ç–∏ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –∫–æ–º–∞–Ω–¥–∏ switch.**

```js
let language = 'Spanish';

switch (language) {
    case 'English': 
        console.log('Hello!');
        break;
    case 'French':
        console.log('Salut');
        break;
    case 'Spanish':
        console.log('Hola!');
        break;
    case 'Polish':
        console.log('Cze≈õƒá!');
        break;
    default:
        console.log(`I don't know this language`);
}
```

–ó–≤–µ—Ä–Ω–∏ —É–≤–∞–≥—É, —â–æ –∫–æ–∂–µ–Ω –≤–∞—Ä—ñ–∞–Ω—Ç –∑–∞–∫—ñ–Ω—á—É—î—Ç—å—Å—è –∫–æ–º–∞–Ω–¥–æ—é **break**. Break –ø–µ—Ä–µ—Ä–∏–≤–∞—î –≤–∏–∫–æ–Ω–∞–Ω–Ω—è –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ switch. –¶–µ –æ–∑–Ω–∞—á–∞—î, —â–æ –ø—Ä–∏ –≤–∏–∫–æ–Ω–∞–Ω–Ω—ñ –±—É–¥—å-—è–∫–æ–≥–æ –∑ –≤–∫–∞–∑–∞–Ω–∏—Ö –≤–∏–ø–∞–¥–∫—ñ–≤ –ø–æ–¥–∞–ª—å—à—ñ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è –≤–∏–∫–æ–Ω—É–≤–∞—Ç–∏—Å—è –Ω–µ –±—É–¥—É—Ç—å. –Ø–∫—â–æ –º–∏ –æ–ø—É—Å—Ç–∏–º–æ —Ü–µ —Å–ª–æ–≤–æ, —Ç–æ –Ω–∞–≤—ñ—Ç—å —è–∫—â–æ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è –±—É–¥–µ —É—Å–ø—ñ—à–Ω–∏–º, –±—É–¥—É—Ç—å –≤–∏–∫–æ–Ω–∞–Ω—ñ –ø–æ–¥–∞–ª—å—à—ñ –ø–µ—Ä–µ–≤—ñ—Ä–∫–∏. –ù–∞—à–∞ —ñ–Ω—Å—Ç—Ä—É–∫—Ü—ñ—è switch –∑–∞–∫—ñ–Ω—á—É—î—Ç—å—Å—è —Å–ø–µ—Ü—ñ–∞–ª—å–Ω–∏–º –≤–∏–ø–∞–¥–∫–æ–º **default**, —è–∫–∏–π –±—É–¥–µ –æ–±—Ä–∞–Ω–æ, –∫–æ–ª–∏ –≤—Å—ñ —ñ–Ω—à—ñ –≤–∞—Ä—ñ–∞–Ω—Ç–∏ –±—É–¥—É—Ç—å –Ω–µ–ø—Ä–∞–≤–∏–ª—å–Ω–∏–º–∏.

### –ó–∞–≤–¥–∞–Ω–Ω—è

–°—Ç–≤–æ—Ä–∏ —É —Å–≤–æ—î–º—É JS-—Ñ–∞–π–ª—ñ –∑–º—ñ–Ω–Ω—É –∑ —ñ–º'—è–º `weather` —ñ –ø—Ä–∏—Å–≤–æ–π—Ç–µ —ó–π –∑–Ω–∞—á–µ–Ω–Ω—è "sun". –ü–æ—Ç—ñ–º, –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—á–∏ –∫–æ–º–∞–Ω–¥—É `switch`, –∑—Ä–æ–±—ñ—Ç—å —Ç–∞–∫, —â–æ–± –≤ –∫–æ–Ω—Å–æ–ª—ñ –∑'—è–≤–∏–≤—Å—è –Ω–∞—Å—Ç—É–ø–Ω–∏–π —Ç–µ–∫—Å—Ç:

**–£–º–æ–≤–∞ if..else... –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î –æ–ø–µ—Ä–∞—Ç–æ—Ä–∏ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è**
